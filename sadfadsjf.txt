        // ========== TYPING EFFECT ==========
        function typeText(element, texts, speed = 100) {
            let textIndex = 0;
            let charIndex = 0;

            function type() {
                if (charIndex < texts[textIndex].length) {
                    element.textContent += texts[textIndex].charAt(charIndex);
                    charIndex++;
                    setTimeout(type, speed);
                } else {
                    charIndex = 0;
                    textIndex++;
                    if (textIndex < texts.length) {
                        setTimeout(() => {
                            element.textContent = '';
                            type();
                        }, 2000);
                    }
                }
            }

            type();
        }
